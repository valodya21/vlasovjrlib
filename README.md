# vlasovjrlib

add to your pow.xml file

```xml
  <repositories>
    <repository>
      <id>vlasovjrlib-mvn-repo</id>
      <url>https://raw.github.com/valodya21/vlasovjrlib/mvn-repo/</url>
      <snapshots>
        <enabled>true</enabled>
        <updatePolicy>always</updatePolicy>
      </snapshots>
    </repository>
  </repositories>

  <dependencies>
    <dependency>
      <groupId>vlasovjr</groupId>
      <artifactId>vlasovjr.lib</artifactId>
      <version>0.0.4a-SNAPSHOT</version>
    </dependency>
  </dependencies>
```
